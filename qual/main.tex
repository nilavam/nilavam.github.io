\pb{Groups}{ 
Let $A$ be the abelian group with generators $x,y,z$ and the relations
$$4x+3y+z=0, x+2y+3z=0, 3x+2y+5z=0$$ Show that $A$ is a cyclic abelian group, and determine its order.
}

%\rule{\textwidth}{2pt}
\pb{Linear Algebra}{ 
Let $A$ be a complex $n \times n$ matrix. Prove that there is an invertible complex $n \times n$ matrix $B$ such that $AB = BA^t$. ($A^t$ is the transpose of $A$.)
}

If $R = Z[x]$, show that the sequence $R\stackrel{f}{\to} R^2\stackrel{g}{\to} R$ is exact, where $ f(a) = (ax, -2a)$ and $g(c,d) = 2c+dx$.
}

%\rule{\textwidth}{2pt}
\pb{Groups}{ 
Let $G$ be a finite simple group. Prove that $G \times G$ has exactly $4$ normal subgroups (including $G \times G$) if and only if $G$ is non-abelian.
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%\rule{\textwidth}{2pt}
\pb{Rings}{ 
Let $R$ be a principal ideal domain and $I,J$ be ideals of $R$. Show that
$I \cap J = IJ$ holds if and only if $I =0$ or $J = 0$ or $I + J = R$.
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%\rule{\textwidth}{2pt}
\pb{Linear Algebra}{ 
Let $A \in M_n(\R)$ be a symmetric matrix with real coefficients. Show that all eigenvalues of $A$ are non-negative if and only if $A = P^T P$ for some matrix $P \in M_n(\R)$.
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%\rule{\textwidth}{2pt}
\pb{Rings}{ 
Let $R$ be an integral domain and $R[x,y,z]$ the polynomial ring in three variables over $R$. Show that $I = \left<x^3 ,y^2, y^3 -z^2y\right> \subseteq R[x, y, z]$ is a prime ideal. \\
Hint: Show that $I$ is the kernel of a ring homomorphism $R[x, y, z] \to R[t]$.
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%\rule{\textwidth}{2pt}
\pb{Linear Algebra}{ 
Let $A$ and $B$ be commuting complex matrices. Assume that $B \notin \C[A]$, that is, $B$ cannot be written as a polynomial in $A$. Show that some eigenspace of $A$ has dimension at least two.
}
