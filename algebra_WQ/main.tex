\begin{itemize}[noitemsep]

\pb{Algebra}{ 
Suppose that $A$ is a not necessarily commutative, finite dimensional associative algebra with a unit over a field $F$ and $P \unlhd A$ is a two-sided ideal such that for $a,b\in A, ab\in P \implies a\in P$ or $bP\in P$. Show that $A/P$ must be a division algebra (i.e. every nonzero element has a multiplicative inverse).
%\import{./soln/S20/}{1.tex}
}

%\rule{\textwidth}{2pt}
\pb{Linear Algebra}{ 
Let $M$ be a matrix with integer entries.
\begin{enumerate}[label=(\alph*)]
\item Prove that the minimal polynomial of $M$ over $\C$ $$f_{\min}(t) = t^k + \sum_{i=0}^{k-1}a_it^i$$ has integer coefficients.
\item Prove that if $M$ is diagonalizable over $\Q$ then there exists an integer $N$ such that the matrix $M \mod p$ is diagonalizable over $\Z/p\Z$ for all $p > N$.
\end{enumerate}
%\import{./soln/S20/}{3.tex}
}

(a) Show that $L$ is not a finitely generated module over $R$.
(b) Show that every finitely generated submodule of $L$ is free with a single generator.

\end{enumerate}
